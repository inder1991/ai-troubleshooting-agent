Replace `_resolve_latest_tag` body:

```python
import re
_SEMVER_RE = re.compile(r"^v(\d+)\.(\d+)\.(\d+)$")


def _resolve_latest_tag(git_url: str) -> str:
    """Pick the highest semver-style v* tag from a remote, or fall back to 'main'.

    B9 (v1.1.1): lexical sort ranks v1.10.0 below v1.2.0. Parse each
    tag as (major, minor, patch) ints and sort numerically. Tags that
    don't match `v\\d+\\.\\d+\\.\\d+` are skipped — pre-release or
    branch-style refs shouldn't be auto-pinned.
    """
    out = subprocess.check_output(
        ["git", "ls-remote", "--tags", git_url], text=True, timeout=30,
    )
    parsed: list[tuple[tuple[int, int, int], str]] = []
    for line in out.splitlines():
        if "refs/tags/" not in line:
            continue
        tag = line.split("refs/tags/")[-1]
        if tag.endswith("^{}"):
            continue
        m = _SEMVER_RE.match(tag)
        if not m:
            continue
        parsed.append((tuple(int(g) for g in m.groups()), tag))
    if not parsed:
        return "main"
    parsed.sort()
    return parsed[-1][1]
```

**Why this matters:** B2 (v1.1.0) put `fcntl.LOCK_EX` around

and both rename onto `.1`, clobbering the first rotation's bytes.

REPO_ROOT = Path(__file__).resolve().parents[2]


def _rotate_worker(log_path_str):
    sys.path.insert(0, str(REPO_ROOT))
    from tools import run_validate
    run_validate.FAILURE_LOG_PATH = Path(log_path_str)
    run_validate.FAILURE_LOG_MAX_BYTES = 100
    run_validate._rotate_failure_log()


def test_concurrent_rotates_dont_lose_entries():
    """B10: 10 concurrent _rotate_failure_log calls must not corrupt
    the rotated .1 file."""
    with tempfile.TemporaryDirectory() as td:
        log = Path(td) / "log.jsonl"
        with log.open("w") as fh:
            for i in range(50):
                fh.write(json.dumps({"i": i}) + "\n")
        size_before = log.stat().st_size
        assert size_before > 100

        with mp.Pool(10) as pool:
            pool.map(_rotate_worker, [str(log)] * 10)

        assert rotated.stat().st_size == size_before, (
            f".1 size {rotated.stat().st_size} != original {size_before}; "
            "concurrent rotates clobbered each other"
        )
        # The live log either doesn't exist (fresh writes hadn't started)
        # or is small (only post-rotate writes if any).

            assert log.stat().st_size < size_before

def _rotate_failure_log() -> None:
    """If FAILURE_LOG_PATH exceeds its size cap, rename to .1.

    B10 (v1.1.1): rotate path is now held under fcntl.LOCK_EX so
    concurrent run_validate processes can't both observe size > cap
    and double-rename, clobbering the first rotation's bytes. We
    re-check size under the lock — another process may have already
    rotated between our exists() and our lock acquisition.
    """
    if not FAILURE_LOG_PATH.exists():
        return
    if not _HAVE_FCNTL:
        try:
            if FAILURE_LOG_PATH.stat().st_size > FAILURE_LOG_MAX_BYTES:
                rotated = FAILURE_LOG_PATH.with_suffix(FAILURE_LOG_PATH.suffix + ".1")
                if rotated.exists():
                    rotated.unlink()
                FAILURE_LOG_PATH.rename(rotated)
        except OSError:
            pass
        return
    try:
        with FAILURE_LOG_PATH.open("a+", encoding="utf-8") as fh:
            _fcntl.flock(fh.fileno(), _fcntl.LOCK_EX)
            try:
                if FAILURE_LOG_PATH.stat().st_size <= FAILURE_LOG_MAX_BYTES:
                    return
                rotated = FAILURE_LOG_PATH.with_suffix(FAILURE_LOG_PATH.suffix + ".1")
                if rotated.exists():
                    try:
                        rotated.unlink()
                    except OSError:
                        pass
                FAILURE_LOG_PATH.rename(rotated)
            finally:
                _fcntl.flock(fh.fileno(), _fcntl.LOCK_UN)
    except OSError:
        pass
```

REPO_ROOT = Path(__file__).resolve().parents[2]
TARGETS = [
    REPO_ROOT / "tools/init_harness.py",
    REPO_ROOT / "tools/sync_harness.py",

GIT_CALL_RE = re.compile(
    r'subprocess\.(?:run|check_call|check_output)\s*\(\s*\[\s*["\']git["\']',
    re.MULTILINE,
)

def test_every_git_subprocess_has_timeout():
    """B13: every subprocess invocation that calls `git` must include
    a `timeout=` argument so a hung remote can't stall bootstrap."""
    offenders = []
    for path in TARGETS:
        text = path.read_text()
        for match in GIT_CALL_RE.finditer(text):
            # Take a 400-char window after the match opening; check for `timeout=`.
            window = text[match.start():match.start() + 600]
            # The window may extend past the call; we only care about whether
            # `timeout=` appears before the matching `)` at depth 0.
            depth = 0
            close = None
            for i, ch in enumerate(window):
                if ch == "(":
                    depth += 1
                elif ch == ")":
                    depth -= 1
                    if depth == 0:
                        close = i
                        break
            call_text = window[: close + 1] if close else window
            if "timeout=" not in call_text:
                line_no = text[: match.start()].count("\n") + 1
                offenders.append(f"{path.name}:{line_no}")
    assert not offenders, (
        f"{len(offenders)} git subprocess call(s) missing timeout=: {offenders}"
    )

def _verify_tag(
    clone_dir: Path, ref: str, trust_fingerprint: str | None = None,
) -> tuple[bool, str]:
    cat = subprocess.run(
        ["git", "cat-file", "-t", ref],
        cwd=clone_dir, capture_output=True, text=True, timeout=10,
    )
    obj_type = (cat.stdout or "").strip()
    if cat.returncode != 0 or obj_type != "tag":
        return False, (...)
    verify = subprocess.run(
        ["git", "verify-tag", "--raw", ref],
        cwd=clone_dir, capture_output=True, text=True, timeout=15,
    )
    if verify.returncode != 0:
        return False, (...)
    if trust_fingerprint:
        # `git verify-tag --raw` emits VALIDSIG <fpr> ... on stderr.
        sig_lines = [l for l in verify.stderr.splitlines() if "VALIDSIG" in l]
        actual_fpr = sig_lines[0].split()[2] if sig_lines else None
        if actual_fpr != trust_fingerprint.upper():
            return False, (
                f"signature is valid but fingerprint {actual_fpr} != "
                f"--trust-key {trust_fingerprint}"
            )
    return True, (verify.stderr or verify.stdout or "").strip()

#!/usr/bin/env python3
"""Q21 — HARNESS_CARD.version must match .harness-version.

Q21.harness-card-version-mismatch — bumping `.harness-version` without
also bumping `HARNESS_CARD.yaml`'s `version` field (or vice versa).
The card's version is hand-maintained; this check is the only thing
that catches drift before consumers see it.
"""
from __future__ import annotations
import sys
from pathlib import Path
import yaml

REPO_ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(REPO_ROOT / ".harness/checks"))
from _common import emit  # noqa: E402

CARD = REPO_ROOT / ".harness/HARNESS_CARD.yaml"
PIN = REPO_ROOT / ".harness-version"


def main() -> int:
    if not CARD.exists() or not PIN.exists():
        return 0   # nothing to compare
    pin = PIN.read_text(encoding="utf-8").strip().lstrip("v")
    try:
        card = yaml.safe_load(CARD.read_text(encoding="utf-8")) or {}
    except yaml.YAMLError as exc:
        emit("WARN", CARD, "harness.unparseable",
             f"could not parse {CARD.name}: {exc}", "fix YAML syntax")
        return 0
    card_v = str(card.get("version", "")).strip()
    if pin != card_v:
        emit("ERROR", CARD, "Q21.harness-card-version-mismatch",
             f"HARNESS_CARD.version={card_v!r} but .harness-version={pin!r}",
             f"bump HARNESS_CARD.yaml `version` to {pin!r} in the same commit",
             line=1)
        return 1
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
