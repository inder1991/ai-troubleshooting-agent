```python
import re
_SEMVER_RE = re.compile(r"^v(\d+)\.(\d+)\.(\d+)$")

def _resolve_latest_tag(git_url: str) -> str:
    """Pick the highest semver-style v* tag from a remote, or fall back to 'main'.

    B9 (v1.1.1): lexical sort ranks v1.10.0 below v1.2.0. We now parse
    each tag as (major, minor, patch) and sort numerically. Tags that
    don't match `v\\d+\\.\\d+\\.\\d+` are skipped — pre-release or
    branch-style refs shouldn't be auto-pinned.
    """
    out = subprocess.check_output(
        ["git", "ls-remote", "--tags", git_url], text=True, timeout=30,
    )
    parsed: list[tuple[tuple[int, int, int], str]] = []
    for line in out.splitlines():
        if "refs/tags/" not in line:
            continue
        tag = line.split("refs/tags/")[-1]
        if tag.endswith("^{}"):       # peeled tag pointer; ignore the duplicate
            continue
        m = _SEMVER_RE.match(tag)
        if not m:
            continue
        parsed.append((tuple(int(g) for g in m.groups()), tag))
    if not parsed:
        return "main"
    parsed.sort()
    return parsed[-1][1]
```

(Note: this also fixes part of B13 — adds a `timeout=30` to the
`ls-remote` subprocess. Keep the change scoped; document the timeout
add in the commit message.)

**Step 3: Implement the fix**

In `tools/run_validate.py`, rewrite `_rotate_failure_log` to take the
exclusive lock on the live log file, re-check size under the lock,
then perform the rename. The lock object only exists while the file
is open — closing it releases the lock.

```python
def _rotate_failure_log() -> None:
    """If FAILURE_LOG_PATH exceeds its size cap, rename to .1.

    B10 (v1.1.1): the rotate path must be held under the same fcntl
    LOCK_EX as appends so concurrent run_validate processes don't both
    observe `size > cap` and double-rename, clobbering the first
    rotation. We open the log read-write, take LOCK_EX, RE-CHECK size
    (the previous holder may have already rotated), then rename.
    """
    if not FAILURE_LOG_PATH.exists():
        return
    if not _HAVE_FCNTL:
        # Windows: best-effort, no lock available (documented).
        try:
            if FAILURE_LOG_PATH.stat().st_size > FAILURE_LOG_MAX_BYTES:
                rotated = FAILURE_LOG_PATH.with_suffix(FAILURE_LOG_PATH.suffix + ".1")
                if rotated.exists():
                    rotated.unlink()
                FAILURE_LOG_PATH.rename(rotated)
        except OSError:
            pass
        return
    try:
        with FAILURE_LOG_PATH.open("a+", encoding="utf-8") as fh:
            _fcntl.flock(fh.fileno(), _fcntl.LOCK_EX)
            try:
                # Re-check under the lock; another process may have
                # already rotated between our exists() and our lock.
                if FAILURE_LOG_PATH.stat().st_size <= FAILURE_LOG_MAX_BYTES:
                    return
                rotated = FAILURE_LOG_PATH.with_suffix(FAILURE_LOG_PATH.suffix + ".1")
                if rotated.exists():
                    try:
                        rotated.unlink()
                    except OSError:
                        pass
                FAILURE_LOG_PATH.rename(rotated)
            finally:
                _fcntl.flock(fh.fileno(), _fcntl.LOCK_UN)
    except OSError:
        # Rotation must never abort validate.
        pass
```
