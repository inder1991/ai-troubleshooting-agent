# Harness Sprint H.3 — Per-Task Implementation Plan (Extract to standalone repo)

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Extract the harness substrate (`.harness/`, `tools/`, the consolidated design + per-task plans, the init templates) from DebugDuck into a standalone `github.com/<owner>/ai-harness` repo so any new project can bootstrap without depending on DebugDuck's working tree. Make DebugDuck itself a *consumer* of the standalone harness, pinned at a tagged version.

**Architecture:** One-time `git filter-repo` (or plain `cp -r`) extraction into a fresh repo, preserving authorship and the seven-sprint plan history. Tag `v1.0`. Update `tools/init_harness.py` to support **two** bootstrap modes: (a) local-source (current — copies from working tree), (b) git-source (clones a pinned ref of the harness repo, then copies). DebugDuck adds `tools/sync_harness.py` that pulls a pinned harness version into its own `.harness/` + `tools/` paths via shallow clone — keeping DebugDuck's harness in lockstep with the canonical repo.

**Tech Stack:** Python 3.14 (stdlib only), git (shell out), `git-filter-repo` (one-time installer; brew install).

```python
#!/usr/bin/env python3
"""Walks the repo and writes the canonical file list for harness extraction.

Output: tools/extraction/manifest.txt — one path per line, sorted, deterministic.
"""
from __future__ import annotations

import sys
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parents[2]
INCLUDE_ROOTS = (
    ".harness",
    "tools",
    "docs/plans/2026-04-26-ai-harness.md",
    "docs/plans/2026-04-26-harness-sprint-h0a-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h0b-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h1a-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h1b-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h1c-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h1d-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h2-tasks.md",
    "docs/plans/2026-04-26-harness-sprint-h3-extract-tasks.md",
    "docs/decisions",
    "tests/harness",
    "CLAUDE.md",
    "AGENTS.md",
    ".cursorrules",
    "Makefile",
    "CONTRIBUTING.md",
)
EXCLUDE_TOKENS = (
    "__pycache__", ".venv", "node_modules", ".pytest_cache",
)


def collect() -> list[str]:
    out: set[str] = set()
    for entry in INCLUDE_ROOTS:
        path = REPO_ROOT / entry
        if not path.exists():
            continue
        if path.is_file():
            out.add(entry)
            continue
        for f in path.rglob("*"):
            if not f.is_file():
                continue
            if any(tok in str(f) for tok in EXCLUDE_TOKENS):
                continue
            out.add(str(f.relative_to(REPO_ROOT)))
    return sorted(out)


def main() -> int:
    manifest = REPO_ROOT / "tools" / "extraction" / "manifest.txt"
    paths = collect()
    manifest.write_text("\n".join(paths) + "\n", encoding="utf-8")
    print(f"[INFO] {manifest}: {len(paths)} entries")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

### Task 5.2: Implement `tools/sync_harness.py`

```python
#!/usr/bin/env python3
"""Pull a pinned ai-harness version into this repo.

Reads .harness-version (single ref string), shallow-clones
github.com/<owner>/ai-harness at that ref into a tempdir, then overlays:
  - .harness/ (entire tree)
  - tools/load_harness.py
  - tools/run_validate.py
  - tools/run_harness_regen.py
  - tools/_session_start_hook.sh
  - tools/install_pre_commit.sh

Idempotent. Local edits to non-overlay files (root CLAUDE.md, Makefile,
tests/harness/, project-specific *_policy.yaml) are preserved.
"""

from __future__ import annotations

import argparse
import shutil
import subprocess
import sys
import tempfile
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parents[1]
DEFAULT_URL = "https://github.com/<owner>/ai-harness.git"
OVERLAY_PATHS = (
    ".harness",
    "tools/load_harness.py",
    "tools/run_validate.py",
    "tools/run_harness_regen.py",
    "tools/_session_start_hook.sh",
    "tools/install_pre_commit.sh",
    "tools/_common.py",
)
PRESERVE_HARNESS_PROJECT_FILES = (
    ".harness-version",  # pin file lives in DebugDuck, not the harness repo
)


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--ref", help="Override .harness-version with this ref.")
    parser.add_argument("--git-url", default=DEFAULT_URL)
    args = parser.parse_args(argv)

    pin_file = REPO_ROOT / ".harness-version"
    if not pin_file.exists() and not args.ref:
        print(f"[ERROR] {pin_file} missing and --ref not supplied", file=sys.stderr)
        return 2
    ref = args.ref or pin_file.read_text(encoding="utf-8").strip()

    tmp = Path(tempfile.mkdtemp(prefix="ai-harness-sync-"))
    try:
        subprocess.check_call(
            ["git", "clone", "--depth", "1", "--branch", ref, args.git_url, str(tmp)],
        )
        for rel in OVERLAY_PATHS:
            src = tmp / rel
            dst = REPO_ROOT / rel
            if not src.exists():
                continue
            if src.is_dir():
                if dst.exists():
                    shutil.rmtree(dst)
                shutil.copytree(src, dst)
            else:
                dst.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(src, dst)
        print(f"[INFO] synced harness ref={ref} into {REPO_ROOT}")
        return 0
    finally:
        shutil.rmtree(tmp, ignore_errors=True)


if __name__ == "__main__":
    sys.exit(main())
```
